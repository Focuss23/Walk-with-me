Letâ€™s visualize the data into histograms

ggplot(data = bbq,
       mapping = aes(
         x = viscosity
       ))+
  geom_histogram(bins = 10, fill = "lightblue")+
  labs(
    y = "Frequency"
  )+
  facet_wrap(~line)+
  theme_minimal()
bbq %>% 
  group_by(line) %>% 
  summarise(
   Mean = mean(viscosity),
    Median = median(viscosity),
    `Standard deviation` = sd(viscosity),
    Kurtosis = kurtosis(viscosity),
    Skewness = skewness(viscosity),
    Range = max(viscosity) - min(viscosity),
    Minimum = min(viscosity),
    Maximum = max(viscosity)
  ) %>% 
  qflextable()
